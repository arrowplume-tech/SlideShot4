      return {
        color: this.rgbToHex(r, g, b),
        transparency: Math.round((1 - a) * 100)
      };
    }
    
    return this.cssToHex(cssColor);
  }

  private convertGradient(cssGradient: string): any {
    // –ü–∞—Ä—Å–∏–º CSS –≥—Ä–∞–¥–∏–µ–Ω—Ç –∏ –∫–æ–Ω–≤–µ—Ä—Ç–∏—Ä—É–µ–º –≤ PowerPoint –≥—Ä–∞–¥–∏–µ–Ω—Ç
    const stops = this.parseGradientStops(cssGradient);
    
    return {
      type: 'gradient',
      colors: stops.map(stop => ({
        color: this.cssToHex(stop.color),
        position: stop.position
      })),
      angle: this.parseGradientAngle(cssGradient)
    };
  }

  private convertLine(cssLine: any): any {
    if (!cssLine) return undefined;
    
    return {
      color: this.cssToHex(cssLine.color),
      width: this.parseSize(cssLine.width),
      dashType: this.convertDashType(cssLine.style)
    };
  }

  private convertShadow(cssShadow: string): any {
    if (!cssShadow || cssShadow === 'none') return undefined;
    
    const parts = cssShadow.split(' ');
    return {
      type: 'outer',
      blur: this.parseSize(parts[2]),
      offset: this.parseSize(parts[0]),
      angle: 45,
      color: '000000',
      opacity: 0.3
    };
  }

function convertColorToPptx(cssColor: string | undefined): any {

function rgbToHex(rgb: string): string {
  const matches = rgb.match(/\d+/g);
  if (!matches || matches.length < 3) return '000000';
  
  const r = parseInt(matches[0]).toString(16).padStart(2, '0');
  const g = parseInt(matches[1]).toString(16).padStart(2, '0');
  const b = parseInt(matches[2]).toString(16).padStart(2, '0');
  
  return (r + g + b).toUpperCase();
}

function extractGradientColors(gradient: string): string[] {
  const colorRegex = /(rgba?\([^)]+\)|#[a-fA-F0-9]{3,6}|\w+)\s+\d+%/g;
  const matches = gradient.match(colorRegex) || [];
  return matches.map(match => match.split(/\s+/)[0]);
}

function extractGradientAngle(gradient: string): number {
  const angleMatch = gradient.match(/(\d+)deg/);
  if (angleMatch) return parseInt(angleMatch[1]);
  
  if (gradient.includes('to right')) return 90;
  if (gradient.includes('to left')) return 270;
  if (gradient.includes('to bottom')) return 180;
  if (gradient.includes('to top')) return 0;

}
